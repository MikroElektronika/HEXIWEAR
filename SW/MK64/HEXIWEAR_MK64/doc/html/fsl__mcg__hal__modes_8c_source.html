77 /*FUNCTION******************************************************************************
78  *
79  * Functon name : CLOCK_HAL_GetMcgMode
80  * Description : This function will check the mcg registers and determine
81  * current MCG mode.
82  *
83  * Return value : mcgMode or error code mcg_modes_t defined in fsl_mcg_hal_modes.h
84  *END*/
85 mcg_modes_t CLOCK_HAL_GetMcgMode(MCG_Type * base)
86 {
87  mcg_modes_t mode = kMcgModeError;
88  mcg_clkout_stat_t clkst = CLOCK_HAL_GetClkOutStat(base);
89  mcg_fll_src_t irefst = CLOCK_HAL_GetFllSrc(base);
90  uint8_t lp = MCG_BRD_C2_LP(base);
91 #if FSL_FEATURE_MCG_HAS_PLL
92  bool pllst = CLOCK_HAL_IsPllSelected(base);
93 #endif
94 
128  switch (clkst)
129  {
130  case kMcgClkOutStatFll:
131 #if FSL_FEATURE_MCG_HAS_PLL
132  if (!pllst)
133 #endif
134  {
135 
136  if (kMcgFllSrcExternal == irefst)
137  {
138  mode = kMcgModeFEE;
139  }
140  else
141  {
142  mode = kMcgModeFEI;
143  }
144  }
145  break;
146  case kMcgClkOutStatInternal:
147  if ((kMcgFllSrcInternal == irefst)
148 #if FSL_FEATURE_MCG_HAS_PLL
149  && (!pllst)
150 #endif
151  )
152  {
153  if (0U == lp)
154  {
155  mode = kMcgModeFBI;
156  }
157  else
158  {
159  mode = kMcgModeBLPI;
160  }
161  }
162  break;
163  case kMcgClkOutStatExternal:
164  if (kMcgFllSrcExternal == irefst)
165  {
166  if (0U == lp)
167  {
168 #if FSL_FEATURE_MCG_HAS_PLL
169  if (pllst)
170  {
171  mode = kMcgModePBE;
172  }
173  else
174 #endif
175  {
176  mode = kMcgModeFBE;
177  }
178  }
179  else
180  {
181  mode = kMcgModeBLPE;
182  }
183  }
184  break;
185 #if FSL_FEATURE_MCG_HAS_PLL
186  case kMcgClkOutStatPll:
187  if ((kMcgFllSrcExternal == irefst) && pllst)
188  {
189  mode = kMcgModePEE;
190  }
191  break;
192 #endif
193  default:
194  break;
195  }
196 
197  return mode;
198 }

221 /*FUNCTION******************************************************************************
222  *
223  * Function name : CLOCK_HAL_SetFeiMode
224  * Description : This function sets MCG to FEI mode.
225  *
226  *END*/
227 mcg_mode_error_t CLOCK_HAL_SetFeiMode(MCG_Type * base,
228  mcg_dco_range_select_t drs,
229  void (* fllStableDelay)(void),
230  uint32_t *outClkFreq)
231 {
232  uint32_t mcgOut;
233 
234  mcg_modes_t mode = CLOCK_HAL_GetMcgMode(base);
235  if (!((kMcgModeFBI | kMcgModeFBE | kMcgModeFEE) & (uint32_t)mode))
236  {
237  return kMcgModeErrModeUnreachable;
238  }
239 
242  mcgOut = CLOCK_HAL_TestFllFreq(base,
243  g_slowInternalRefClkFreq,
244  kMcgDmx32Default,
245  drs);
246  if (0U == mcgOut)
247  {
248  return kMcgModeErrFllRefRange;
249  }
250 
252  MCG_BWR_C4_DMX32(base, kMcgDmx32Default);
253 
255  MCG_WR_C1(base, ((MCG_RD_C1(base) & ~(MCG_C1_CLKS_MASK | MCG_C1_IREFS_MASK)))
256  | (MCG_C1_CLKS(kMcgClkOutSrcOut)
257  | MCG_C1_IREFS(kMcgFllSrcInternal)));
260  while (CLOCK_HAL_GetFllSrc(base) != kMcgFllSrcInternal) {}
261 
262  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatFll) {}
264  MCG_WR_C4_DRST_DRS(base, drs);
265 
267  while (MCG_BRD_C4_DRST_DRS(base) != drs) {}
268 
270  fllStableDelay();
271 
272  *outClkFreq = mcgOut;
273 
274  return kMcgModeErrNone;
275 }

277 /*FUNCTION******************************************************************************
278  *
279  * Functon name : CLOCK_HAL_SetFeeMode
280  * Description : This function sets MCG to FEE mode.
281  *
282  *END*/
283 mcg_mode_error_t CLOCK_HAL_SetFeeMode(MCG_Type * base,
284  mcg_oscsel_select_t oscselVal,
285  uint8_t frdivVal,
286  mcg_dmx32_select_t dmx32,
287  mcg_dco_range_select_t drs,
288  void (* fllStableDelay)(void),
289  uint32_t *outClkFreq)
290 {
291  uint32_t mcgOut, extFreq;
292 
293  mcg_modes_t mode = CLOCK_HAL_GetMcgMode(base);
294  if (!((kMcgModeFBI | kMcgModeFBE | kMcgModeFEI) & (uint32_t)mode))
295  {
296  return kMcgModeErrModeUnreachable;
297  }
298 
299  extFreq = CLOCK_HAL_TestOscFreq(base, oscselVal);
300 
301  if (0U == extFreq)
302  {
303  return kMcgModeErrOscFreqRange;
304  }
305 
307  extFreq = CLOCK_HAL_TestFllExternalRefFreq(base,
308  extFreq,
309  frdivVal,
310  (osc_range_t)MCG_BRD_C2_RANGE(base),
311  oscselVal);
312  if ((extFreq < kMcgConstant31250) || (extFreq > kMcgConstant39063))
313  {
314  return kMcgModeErrFllFrdivRange;
315  }
316 
318  mcgOut = CLOCK_HAL_TestFllFreq(base, extFreq, dmx32, drs);
319  if (0U == mcgOut)
320  {
321  return kMcgModeErrFllRefRange;
322  }
323 
325 #if FSL_FEATURE_MCG_USE_OSCSEL
326  MCG_BWR_C7_OSCSEL(base, oscselVal);
327 #endif
328 
330  MCG_WR_C1(base, (MCG_RD_C1(base) & ~(MCG_C1_CLKS_MASK | MCG_C1_FRDIV_MASK | MCG_C1_IREFS_MASK))
331  | (MCG_C1_CLKS(kMcgClkOutSrcOut)
332  | MCG_C1_FRDIV(frdivVal)
333  | MCG_C1_IREFS(kMcgFllSrcExternal)));
334 #if FSL_FEATURE_MCG_USE_OSCSEL
335  if (kMcgOscselOsc == oscselVal)
336 #endif
337  {
338  if (MCG_BRD_C2_EREFS(base))
339  {
340  while(!CLOCK_HAL_IsOsc0Stable(base)){}
341  }
342  }
343 
345  while (CLOCK_HAL_GetFllSrc(base) != kMcgFllSrcExternal) {}
346 
348  MCG_WR_C4(base, (MCG_RD_C4(base)
349  & ~(MCG_C4_DMX32_MASK |
350  MCG_C4_DRST_DRS_MASK))
351  | (MCG_C4_DMX32(dmx32) |
352  MCG_C4_DRST_DRS(drs)));
353 
355  while (MCG_BRD_C4_DRST_DRS(base) != drs) {}
356 
358  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatFll) {}
359 
361  fllStableDelay();
362 
363  *outClkFreq = mcgOut;
364 
365  return kMcgModeErrNone;
366 }

370  * Functon name : CLOCK_HAL_SetFbiMode

374 mcg_mode_error_t CLOCK_HAL_SetFbiMode(MCG_Type * base,
375  mcg_dco_range_select_t drs,
376  mcg_irc_mode_t ircSelect,
377  uint8_t fcrdivVal,
378  void (* fllStableDelay)(void),
379  uint32_t *outClkFreq)
380 {
381  mcg_modes_t mode = CLOCK_HAL_GetMcgMode(base);
382 
383  if (!((kMcgModeFEI | kMcgModeFBE | kMcgModeFEE | kMcgModeBLPI) & (uint32_t)mode))
384  {
385  return kMcgModeErrModeUnreachable;
386  }
387 
389  MCG_BWR_C4_DMX32(base, kMcgDmx32Default);
390 
392  MCG_WR_C1(base, (MCG_RD_C1(base) & ~(MCG_C1_CLKS_MASK | MCG_C1_IREFS_MASK))
393  | (MCG_C1_CLKS(kMcgClkOutSrcInternal)
394  | MCG_C1_IREFS(kMcgFllSrcInternal)));
397  while (CLOCK_HAL_GetFllSrc(base) != kMcgFllSrcInternal) {}
398 
399  if (kMcgIrcFast == ircSelect)
400  {
402  CLOCK_HAL_UpdateFastClkInternalRefDiv(base, fcrdivVal);
403  }
405  MCG_WR_C2(base, (MCG_RD_C2(base) & ~(MCG_C2_LP_MASK | MCG_C2_IRCS_MASK))
406  | MCG_C2_IRCS(ircSelect));
407 
408  while (CLOCK_HAL_GetInternalRefClkMode(base) != ircSelect){}
409 
410  MCG_WR_C4_DRST_DRS(base, drs);
411 
413  while (MCG_BRD_C4_DRST_DRS(base) != drs) {}
414 
415  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatInternal){}
416 
418  fllStableDelay();
419 
421  if (ircSelect == kMcgIrcSlow)
422  {
423  *outClkFreq = g_slowInternalRefClkFreq;
424  }
425  else
426  {
427  *outClkFreq = (g_fastInternalRefClkFreq >> fcrdivVal);
428  }
429 
430  return kMcgModeErrNone;
431 }

452  | kMcgModeFBE
453  | kMcgModeFEE
454  | kMcgModeBLPE
455 #if FSL_FEATURE_MCG_HAS_PLL
456  | kMcgModePBE
457 #endif
458  ) & (uint32_t)mode))
459  {
460  return kMcgModeErrModeUnreachable;

462 
463  extFreq = CLOCK_HAL_TestOscFreq(base, oscselVal);
464 
465  if (0U == extFreq)
466  {
467  return kMcgModeErrOscFreqRange;
468  }
469 
470 #if FSL_FEATURE_MCG_HAS_PLL
471 
472  MCG_BWR_C6_PLLS(base, 0U);

474 #endif
475 
477  MCG_BWR_C2_LP(base, 0U);
478 
479 #if FSL_FEATURE_MCG_USE_OSCSEL
480  if (kMcgModeFEE != mode)
481  {
482  MCG_BWR_C7_OSCSEL(base, oscselVal);
483  }
484 #endif
485 
487  MCG_WR_C1(base, (MCG_RD_C1(base) & ~(MCG_C1_CLKS_MASK | MCG_C1_FRDIV_MASK | MCG_C1_IREFS_MASK))
488  | (MCG_C1_CLKS(kMcgClkOutSrcExternal)

490  | MCG_C1_IREFS(kMcgFllSrcExternal)));
493  while (kMcgFllSrcExternal != CLOCK_HAL_GetFllSrc(base)) {}
494 
495 #if FSL_FEATURE_MCG_USE_OSCSEL
496  if (kMcgModeFEE == mode)
497  {
498  MCG_BWR_C7_OSCSEL(base, oscselVal);
499  }
500  if (kMcgOscselOsc == oscselVal)
501 #endif
502  {
503  if (MCG_BRD_C2_EREFS(base))
504  {
505  while(!CLOCK_HAL_IsOsc0Stable(base)){}
506  }
507  }
508 
510  MCG_WR_C4(base, (MCG_RD_C4(base)
511  & ~(MCG_C4_DMX32_MASK |
512  MCG_C4_DRST_DRS_MASK))
513  | (MCG_C4_DMX32(dmx32) |
514  MCG_C4_DRST_DRS(drs)));
515 
517  while (MCG_BRD_C4_DRST_DRS(base) != drs) {}
518 
520  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatExternal) {}
521 
523  fllStableDelay();
524 
526  *outClkFreq = extFreq;
527 
528  return kMcgModeErrNone;
529 }

620 mcg_mode_error_t CLOCK_HAL_SetPbeMode(MCG_Type * base,
621  mcg_oscsel_select_t oscselVal,
622  mcg_pll_clk_select_t pllcsSelect,
623  uint8_t prdivVal,
624  uint8_t vdivVal,
625  uint32_t *outClkFreq)
626 #if FSL_FEATURE_MCG_HAS_PLL1
627 {
628  uint32_t extFreq, pllFreq;
629 
630  mcg_modes_t mode = CLOCK_HAL_GetMcgMode(base);
631  if (!((kMcgModeFBE |
632  kMcgModePEE |
633  kMcgModePBE |
634  kMcgModeBLPE) & (uint32_t)mode));
635  {
636  return kMcgModeErrModeUnreachable;
637  }
638 
640  extFreq = CLOCK_HAL_TestOscFreq(base, oscselVal);
641 
642  if (0U == extFreq)
643  {
644  return kMcgModeErrOscFreqRange;
645  }
646 
647  if (pllcsSelect == kMcgPllcsSelectPll1)
648  {
649  pllFreq = CLOCK_HAL_GetPll1RefFreq(base);
650  }
651  else
652  {
653  pllFreq = CLOCK_HAL_GetPll0RefFreq(base);
654  }
655 
656  pllFreq = pllFreq / (prdivVal + FSL_FEATURE_MCG_PLL_PRDIV_BASE);
657 
658  if ((pllFreq < FSL_FEATURE_MCG_PLL_REF_MIN) ||
659  (pllFreq > FSL_FEATURE_MCG_PLL_REF_MAX))
660  {
661  return kMcgErrPllPrdivRange;
662  }
663 
665  MCG_BWR_C2_LP(base, 0U);
666 
668  MCG_BWR_C1_CLKS(base, kMcgClkOutSrcExternal);
669 
671  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatExternal) {}
672 
673  CLOCK_HAL_PrepareOsc(base, oscselVal);
674 
676  MCG_BWR_C6_PLLS(base, 0U);
677  // wait for PLLST status bit to clear.
678  while ((CLOCK_HAL_IsPllSelected(base) != false)) {}
679 
681  if (pllcsSelect == kMcgPllcsSelectPll1)
682  {
683  MCG_BWR_C11_PLLCLKEN1(base, 0U);
684  MCG_WR_C11_PRDIV1(base, prdivVal);
685  MCG_WR_C12_VDIV1(base, vdivVal);
686  }
687  else
688  {
689  MCG_BWR_C5_PLLCLKEN0(base, 0U);
690  MCG_WR_C5_PRDIV0(base, prdivVal);
691  MCG_WR_C6_VDIV0(base, vdivVal);
692  }
693 
694  CLOCK_HAL_SetPllcs(pllcsSelect);
695  while (pllcsSelect != CLOCK_HAL_GetPllClkSelStatMode(base)) {}
696 
698  MCG_BWR_C6_PLLS(base, 1U);
699  // wait for PLLST status bit to set
700  while ((CLOCK_HAL_IsPllSelected(base) != true)) {}
701 
703  if (pllcsSelect == kMcgPllcsSelectPll1)
704  {
705  while (!CLOCK_HAL_IsPll1Locked(base)) {}
706  }
707  else
708  {
709  while (!CLOCK_HAL_IsPll0Locked(base)) {}
710  }
711 
712  *outClkFreq = extFreq;
713  return kMcgModeErrNone;
714 }
715 #else
716 {
717  uint32_t extFreq, pllRefFreq;
718 
719  mcg_modes_t mode = CLOCK_HAL_GetMcgMode(base);
720  if (!((kMcgModeFBE | kMcgModePEE | kMcgModeBLPE) & (uint32_t)mode))
721  {
722  return kMcgModeErrModeUnreachable;
723  }
724 
726  extFreq = CLOCK_HAL_TestOscFreq(base, oscselVal);
727 
728  if (0U == extFreq)
729  {
730  return kMcgModeErrOscFreqRange;
731  }
732 
733 #if FSL_FEATURE_MCG_HAS_EXTERNAL_PLL
734 
735  CLOCK_HAL_SetPllClkSelMode(base, pllcsSelect);
736  while (pllcsSelect != CLOCK_HAL_GetPllClkSelMode(base)) {}
737 
738  if (pllcsSelect == kMcgPllClkSelPll0)
739  {
740 #endif
741 
742  pllRefFreq = extFreq / (prdivVal + FSL_FEATURE_MCG_PLL_PRDIV_BASE);
743 
744  if ((pllRefFreq < FSL_FEATURE_MCG_PLL_REF_MIN) ||
745  (pllRefFreq > FSL_FEATURE_MCG_PLL_REF_MAX))
746  {
747  return kMcgModeErrPllPrdivRange;
748  }
749 #if FSL_FEATURE_MCG_HAS_EXTERNAL_PLL
750  }
751 #endif
752 
753  MCG_BWR_C2_LP(base, 0U);
754 
756  MCG_BWR_C1_CLKS(base, kMcgClkOutSrcExternal);
757 
759  while (CLOCK_HAL_GetClkOutStat(base) != kMcgClkOutStatExternal) {}
760 
761  CLOCK_HAL_PrepareOsc(base, oscselVal);
762 
763  // Disable PLL first, then configure PLL.
764  CLOCK_HAL_SetPll0EnableCmd(base, false);
765  MCG_BWR_C6_PLLS(base, 0U);
766 
767  // wait for PLLST status bit to clear.
