71 void arm_shift_q7(
72  q7_t * pSrc,
73  int8_t shiftBits,
74  q7_t * pDst,
75  uint32_t blockSize)
76 {
77  uint32_t blkCnt;
78  uint8_t sign;
80 #ifndef ARM_MATH_CM0_FAMILY
81 
83  q7_t in1;
84  q7_t in2;
85  q7_t in3;
86  q7_t in4;
89  /*loop Unrolling */
90  blkCnt = blockSize >> 2u;
91 
93  sign = (shiftBits & 0x80);
94 
96  if(sign == 0u)
97  {
100  while(blkCnt > 0u)
101  {
104  in1 = *pSrc;
105  in2 = *(pSrc + 1);
106  in3 = *(pSrc + 2);
107  in4 = *(pSrc + 3);
108 
110  *__SIMD32(pDst)++ = __PACKq7(__SSAT((in1 << shiftBits), 8),
111  __SSAT((in2 << shiftBits), 8),
112  __SSAT((in3 << shiftBits), 8),
113  __SSAT((in4 << shiftBits), 8));
115  pSrc += 4u;
116 
118  blkCnt--;
119  }
120 
123  blkCnt = blockSize % 0x4u;
124 
125  while(blkCnt > 0u)
126  {
129  *pDst++ = (q7_t) __SSAT((*pSrc++ << shiftBits), 8);
130 
132  blkCnt--;
133  }
134  }
135  else
136  {
137  shiftBits = -shiftBits;
140  while(blkCnt > 0u)
141  {
144  in1 = *pSrc;
145  in2 = *(pSrc + 1);
146  in3 = *(pSrc + 2);
147  in4 = *(pSrc + 3);
148 
150  *__SIMD32(pDst)++ = __PACKq7((in1 >> shiftBits), (in2 >> shiftBits),
151  (in3 >> shiftBits), (in4 >> shiftBits));
152 
153 
154  pSrc += 4u;
155 
157  blkCnt--;
158  }
159 
162  blkCnt = blockSize % 0x4u;
163 
164  while(blkCnt > 0u)
165  {
168  in1 = *pSrc++;
169  *pDst++ = (in1 >> shiftBits);
170 
172  blkCnt--;
173  }
174  }
175 
176 #else
177 
181  sign = (shiftBits & 0x80);
182 
184  if(sign == 0u)
185  {
187  blkCnt = blockSize;
188 
189  while(blkCnt > 0u)
190  {
193  *pDst++ = (q7_t) __SSAT(((q15_t) * pSrc++ << shiftBits), 8);
194 
196  blkCnt--;
197  }
198  }
199  else
200  {
202  blkCnt = blockSize;
203 
204  while(blkCnt > 0u)
205  {
208  *pDst++ = (*pSrc++ >> -shiftBits);
209 
211  blkCnt--;
212  }
213  }
214 
215 #endif /* #ifndef ARM_MATH_CM0_FAMILY */
216 }
