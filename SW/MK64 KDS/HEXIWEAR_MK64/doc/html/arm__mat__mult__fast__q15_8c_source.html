91  q31_t sum;
92  q15_t *pSrcBT = pState;
93  q15_t *pInA = pSrcA->pData;
94  q15_t *pInB = pSrcB->pData;
95  q15_t *px;
96  uint16_t numRowsA = pSrcA->numRows;
97  uint16_t numColsB = pSrcB->numCols;
98  uint16_t numColsA = pSrcA->numCols;
99  uint16_t numRowsB = pSrcB->numRows;
100  uint16_t col, i = 0u, row = numRowsB, colCnt;
101  arm_status status;
103 #ifndef UNALIGNED_SUPPORT_DISABLE
104 
105  q31_t in;
106  q31_t inA1, inA2, inB1, inB2;
107 
108 #else
109 
110  q15_t in;
111  q15_t inA1, inA2, inB1, inB2;
112 
113 #endif /* #ifndef UNALIGNED_SUPPORT_DISABLE */
114 
115 #ifdef ARM_MATH_MATRIX_CHECK
116 
117  if((pSrcA->numCols != pSrcB->numRows) ||
118  (pSrcA->numRows != pDst->numRows) || (pSrcB->numCols != pDst->numCols))
119  {
121  status = ARM_MATH_SIZE_MISMATCH;
122  }
123  else
124 #endif
125  {
127  do
128  {
130  col = numColsB >> 2;
131 
133  px = pSrcBT + i;
134 
137  while(col > 0u)
138  {
139 #ifndef UNALIGNED_SUPPORT_DISABLE
140 
141  in = *__SIMD32(pInB)++;
142 
144 #ifndef ARM_MATH_BIG_ENDIAN
145 
146  *px = (q15_t) in;
147 
148 #else
149 
150  *px = (q15_t) ((in & (q31_t) 0xffff0000) >> 16);
151 
152 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
153 
155  px += numRowsB;
156 
158 #ifndef ARM_MATH_BIG_ENDIAN
159 
160  *px = (q15_t) ((in & (q31_t) 0xffff0000) >> 16);
161 
162 #else
163 
164  *px = (q15_t) in;
165 
166 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
167 
169  px += numRowsB;
170 
172  in = *__SIMD32(pInB)++;
173 
175 #ifndef ARM_MATH_BIG_ENDIAN
176 
177  *px = (q15_t) in;
178 
179 #else
180 
181  *px = (q15_t) ((in & (q31_t) 0xffff0000) >> 16);
182 
183 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
184 
186  px += numRowsB;
187 
190 #ifndef ARM_MATH_BIG_ENDIAN
191 
192  *px = (q15_t) ((in & (q31_t) 0xffff0000) >> 16);
193 
194 #else
195 
196  *px = (q15_t) in;
197 
198 #endif /* #ifndef ARM_MATH_BIG_ENDIAN */
199 
200 #else
201 
203  in = *pInB++;
204 
206  *px = in;
207 
209  px += numRowsB;
210 
212  in = *pInB++;
213 
215  *px = in;
216 
218  px += numRowsB;
219 
221  in = *pInB++;
222 
224  *px = in;
225 
227  px += numRowsB;
228 
230  in = *pInB++;
231 
233  *px = in;
234 
235 #endif /* #ifndef UNALIGNED_SUPPORT_DISABLE */
236 
238  px += numRowsB;
239 
241  col--;
242  }
243 
246  col = numColsB % 0x4u;
247 
248  while(col > 0u)
249  {
251  *px = *pInB++;
252 
254  px += numRowsB;
255 
257  col--;
258  }
259 
260  i++;
261 
263  row--;
264 
265  } while(row > 0u);
266 
268  row = numRowsA;
269  i = 0u;
270  px = pDst->pData;
271 
274  do
275  {
277  col = numColsB;
278 
281  pInB = pSrcBT;
282 
284  do
285  {
287  sum = 0;
288 
290  colCnt = numColsA >> 2;
291 
293  pInA = pSrcA->pData + i;
294 
296  while(colCnt > 0u)
297  {
299 #ifndef UNALIGNED_SUPPORT_DISABLE
300 
301  inA1 = *__SIMD32(pInA)++;
302  inB1 = *__SIMD32(pInB)++;
303  inA2 = *__SIMD32(pInA)++;
304  inB2 = *__SIMD32(pInB)++;
305 
306  sum = __SMLAD(inA1, inB1, sum);
307  sum = __SMLAD(inA2, inB2, sum);
308 
309 #else
310 
311  inA1 = *pInA++;
312  inB1 = *pInB++;
313  inA2 = *pInA++;
314  sum += inA1 * inB1;
315  inB2 = *pInB++;
316 
317  inA1 = *pInA++;
318  inB1 = *pInB++;
319  sum += inA2 * inB2;
320  inA2 = *pInA++;
321  inB2 = *pInB++;
322 
323  sum += inA1 * inB1;
324  sum += inA2 * inB2;
325 
326 #endif /* #ifndef UNALIGNED_SUPPORT_DISABLE */
327 
329  colCnt--;
330  }
331 
333  colCnt = numColsA % 0x4u;
334 
335  while(colCnt > 0u)
336  {
338  sum += (q31_t) (*pInA++) * (*pInB++);
339 
340  colCnt--;
341  }
342 
344  *px = (q15_t) (sum >> 15);
345  px++;
346 
348  col--;
349 
350  } while(col > 0u);
351 
352  i = i + numColsA;
353 
355  row--;
356 
357  } while(row > 0u);
358 
360  status = ARM_MATH_SUCCESS;
361  }
362 
364  return (status);
365 }
